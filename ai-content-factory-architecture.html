LangGraph TypedState Schema
from langgraph.graph import StateGraph, START, END
from typing import TypedDict, Annotated, List, Optional

class ContentFactoryState(TypedDict):
  topic: str                         # Input
  research_brief: Optional[str]      # Research Agent output
  platform_drafts: Optional[dict]   # {linkedin, substack, medium, x, instagram}
  image_prompts: Optional[List[str]]# Visual Agent output
  seo_data: Optional[dict]          # {keywords, hooks, hashtags}
  hitl_approved: Optional[dict]    # Human decisions per platform
  hitl_edits: Optional[dict]       # Human-edited content
  published_results: Optional[dict]# {platform: post_id}
  error: Optional[str]              # Error propagation
  run_id: str                       # Trace ID for LangSmith

builder = StateGraph(ContentFactoryState)
builder.add_node("research", research_agent)
builder.add_node("content", content_agent)
builder.add_node("visual", visual_agent)  # parallel
builder.add_node("seo", seo_agent)        # parallel
builder.add_node("hitl", human_review_node, interrupt_before=True)
builder.add_node("publish", publisher_agent)
graph = builder.compile(checkpointer=RedisCheckpointer())
